    function loadasync(url){
        //找到head标签 取第一个
        var head = document.getElementsByTagName("head")[0];
        //创建一个script标签
        var script = document.createElement("script");
        //设置src 路径
        script.src = url ;
        //给head标签 增加一个子标签
        head.appendChild(script);
    }
    </script>
